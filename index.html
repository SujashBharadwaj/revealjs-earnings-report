import pandas as pd

# Simplified margin calc
def gross_margin(revenue, cogs):
    return (revenue - cogs) / revenue

df = pd.DataFrame({
    "revenue": [1240, 1185, 1110],
    "cogs":    [519,  518,  505]
})

df["gm"] = df.apply(lambda r: gross_margin(r["revenue"], r["cogs"]), axis=1)
print(df["gm"].mean())
